local player = game.Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")
local RunService = game:GetService("RunService")

-- Ẩn tên
local function hideName(name)
	if #name <= 2 then return name end
	return string.sub(name,1,2) .. string.rep("*", #name-2)
end

local screenGui = Instance.new("ScreenGui", playerGui)

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0,220,0,80)
frame.Position = UDim2.new(0.5,-110,0,40)
frame.BackgroundTransparency = 1
frame.Parent = screenGui

-- TÊN
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,-10,0,30)
title.Position = UDim2.new(0,5,0,5)
title.BackgroundTransparency = 1
title.Text = "tên: " .. hideName(player.Name)
title.TextSize = 37
title.TextXAlignment = Enum.TextXAlignment.Left
title.Font = Enum.Font.SourceSansBold
title.Parent = frame

-- ĐƠN
local inputBox = Instance.new("TextBox")
inputBox.Size = UDim2.new(1,-10,0,30)
inputBox.Position = UDim2.new(0,5,0,40)
inputBox.BackgroundTransparency = 1
inputBox.Text = "đơn: "
inputBox.TextSize = 32
inputBox.TextXAlignment = Enum.TextXAlignment.Left
inputBox.Font = Enum.Font.SourceSans
inputBox.Parent = frame

-- KHÓA "đơn:"
inputBox:GetPropertyChangedSignal("Text"):Connect(function()
	if string.sub(inputBox.Text,1,5) ~= "đơn: " then
		local content = inputBox.Text:gsub("^đơn:%s*", "")
		inputBox.Text = "đơn: " .. content
	end
	inputBox.CursorPosition = #inputBox.Text + 1
end)

-- 🌈 HIỆU ỨNG CẦU VỒNG
local hue = 0

RunService.RenderStepped:Connect(function()
	hue = hue + 0.005
	if hue > 1 then hue = 0 end
	
	local color = Color3.fromHSV(hue,1,1)
	
	title.TextColor3 = color
	inputBox.TextColor3 = color
end)
